// Compile and link to toy execute file. -rdynamic for extern "C" putchard(x) define.
clang++ -g toy.cpp -rdynamic `llvm-config --cppflags --ldflags --libs core jit native` -O3 -o toy

// Compile to obj
clang++ -c toy.cpp `llvm-config --cppflags` -O3 -o toy.o

// Link .o to executable
clang++ toy.o -rdynamic `llvm-config --ldflags --libs core jit native` -O3 -o toy

// Compile to bitcode (.bc), -emit-llvm: meaning compile to .bc
clang++ -c toy.cpp `llvm-config --cppflags` -emit-llvm -O3 -o toy.bc

// Link .bc to executable file
clang++ toy.bc -rdynamic `llvm-config --ldflags --libs core jit native` -O3 -o toy

LLVM toolchain support as a plug-in for Eclipse CDT
